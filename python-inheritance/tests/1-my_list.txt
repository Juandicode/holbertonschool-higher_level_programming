>>> import os, importlib.util
>>> file_path = os.path.join(os.getcwd(), 'my_list_1_7.py')
>>> spec = importlib.util.spec_from_file_location("my_list", file_path)
>>> module = importlib.util.module_from_spec(spec)
>>> spec.loader.exec_module(module)
>>> MyList = module.MyList

>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[1, 4, 2, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[1, 4, 2, 3, 5]

>>> other = MyList()
>>> other.append(-10)
>>> other.append(3)
>>> other.append(0)
>>> other.append(-1)
>>> print(other)
[-10, 3, 0, -1]
>>> other.print_sorted()
[-10, -1, 0, 3]
>>> print(other)
[-10, 3, 0, -1]

>>> empty = MyList()
>>> print(empty)
[]
>>> empty.print_sorted()
[]
>>> print(empty)
[]
